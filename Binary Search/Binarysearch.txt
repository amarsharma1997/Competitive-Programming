#include <bits/stdc++.h>
using namespace std;
int arr[10007];//

int binarysearch(int l,int r,int fnd)//This function returns the index where we get value fnd it returns -1 if not found.....
{
    //We are having the array in increasing order.....
    int mid;
    while(l<=r)
    {
        mid=(l+r)/2;
        if(arr[mid]==fnd)
        {
            return mid;
        }
        else if(arr[mid]>fnd)
        {
            r=mid-1;
        }
        else
        {
            l=mid+1;
        }
    }
    return -1;
};

int main()
{
    int n;//n is the number of elements in the array.....
    cin>>n;
    int i;
    for(i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    int fnd;
    cin>>fnd;
    cout<<binarysearch(0,n-1,fnd);
    return 0;
}
